```java
public class Main {
	public static ArrayList<Integer>[] numbers;
	public static boolean[] visited;
	public static int a, b, c, answer = 0;

	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		a = Integer.parseInt(st.nextToken());
		b = Integer.parseInt(st.nextToken());
		c = Integer.parseInt(st.nextToken());
		visited = new boolean[a + 1];
		numbers = (ArrayList<Integer>[]) new ArrayList[a + 1];

		for (int i = 1; i <= a; i++)
			numbers[i] = new ArrayList<Integer>();

		for (int i = 0; i < b; i++) {
			String str = br.readLine();
			int u = Integer.parseInt(str.split(" ")[0]);
			int v = Integer.parseInt(str.split(" ")[1]);
			numbers[u].add(v);
		}

		for (int i = 0; i < c; i++) {
			int start = Integer.parseInt(br.readLine());
			dfs(start);
		}
		answer();
	}

	public static void dfs(int start) {
		if (visited[start])
			return;
		visited[start] = true;
		for (int i : numbers[start])
			if (!visited[i])
				dfs(i);
	}

	public static void answer() {
		for (boolean tf : visited)
			if (tf) answer++;
		System.out.println(answer);
	}
}
```
